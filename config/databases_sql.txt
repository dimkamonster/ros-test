table users
CREATE TABLE `rosberry`.`users` ( `id` INT NOT NULL AUTO_INCREMENT , `username` VARCHAR(254) NOT NULL , `email` VARCHAR(254) NOT NULL COMMENT 'users email unique' , `password` VARCHAR(254) NOT NULL COMMENT 'users password' , `avatar` VARCHAR(254) NULL COMMENT 'hash to ava' , `age` TINYINT NULL , `country` VARCHAR(254) NULL , `lat` DECIMAL NULL , `lon` DECIMAL NULL , `access_token` VARCHAR(254) NULL COMMENT 'access token after auth valid' , `access_valid_till` INT NULL COMMENT 'timestamp after access token expired' , `csrf` VARCHAR(254) NULL COMMENT 'x-csrf token for each operation' , `csrf_valid_till` INT NULL COMMENT 'timestamp after csrf token expired' , `updated_at` TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP , `created_at` TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP , PRIMARY KEY (`id`), UNIQUE `u_email` (`email`)) ENGINE = InnoDB CHARSET=utf8mb4 COLLATE utf8mb4_general_ci;
